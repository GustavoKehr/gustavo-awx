	chrony.conf.j2
		# Managed by Ansible - baseline_system
#
# Use public servers from the pool.ntp.org project.
# Please consider joining the pool (https://www.pool.ntp.org/join.html).

{% for s in chrony_servers %}
server {{ s }} iburst
{% endfor %}

# Use NTP servers from DHCP.
sourcedir /run/chrony-dhcp

# Record the rate at which the system clock gains/losses time.
driftfile /var/lib/chrony/drift

# Allow the system clock to be stepped in the first three updates
# if its offset is larger than 1 second.
# makestep 1.0 3

# Enable kernel synchronization of the real-time clock (RTC).
rtcsync

# Enable hardware timestamping on all interfaces that support it.
#hwtimestamp *

# Increase the minimum number of selectable sources required to adjust
# the system clock.
#minsources 2

# Allow NTP client access from local network.
#allow 192.168.0.0/16

# Serve time even if not synchronized to a time source.
#local stratum 10

# Require authentication (nts or key option) for all NTP sources.
#authselectmode require

# Specify file containing keys for NTP authentication.
keyfile /etc/chrony.keys

# Save NTS keys and cookies.
ntsdumpdir /var/lib/chrony

# Insert/delete leap seconds by slewing instead of stepping.
leapsecmode slew

# Get TAI-UTC offset and leap seconds from the system tz database.
leapsectz right/UTC

# Specify directory for log files.
logdir /var/log/chrony

# Select which information is logged.
#log measurements statistics tracking

# 4.0 - Backup date (necessário)

- name: "4.0 - Define backup_date"

#####################################
# 4.1 Política de Senhas – /etc/login.defs (PROD-safe)
# - Backup só se houver mudança
# - Sem backup: true em cada lineinfile
# - Idempotente
#####################################

- name: "4.1 - Stat /etc/login.defs"
  ansible.builtin.stat:
    path: /etc/login.defs
  register: login_defs_stat
  tags: [hardening_security, login_defs]

- name: "4.1 - Ensure PASS_MAX_DAYS 90"
  ansible.builtin.lineinfile:
    path: /etc/login.defs
    regexp: '^\s*PASS_MAX_DAYS\s+'
    line: 'PASS_MAX_DAYS   90'
    create: false
  register: ld_max
  when: login_defs_stat.stat.exists
  tags: [hardening_security, login_defs]

- name: "4.1 - Ensure PASS_MIN_DAYS 7"
  ansible.builtin.lineinfile:
    path: /etc/login.defs
    regexp: '^\s*PASS_MIN_DAYS\s+'
    line: 'PASS_MIN_DAYS   7'
    create: false
  register: ld_min
  when: login_defs_stat.stat.exists
  tags: [hardening_security, login_defs]

- name: "4.1 - Ensure PASS_WARN_AGE 7"
  ansible.builtin.lineinfile:
    path: /etc/login.defs
    regexp: '^\s*PASS_WARN_AGE\s+'
    line: 'PASS_WARN_AGE   7'
    create: false
  register: ld_warn
  when: login_defs_stat.stat.exists
  tags: [hardening_security, login_defs]

- name: "4.1 - Ensure PASS_MIN_LEN 8"
  ansible.builtin.lineinfile:
    path: /etc/login.defs
    regexp: '^\s*PASS_MIN_LEN\s+'
    line: 'PASS_MIN_LEN    8'
    create: false
  register: ld_len
  when: login_defs_stat.stat.exists
  tags: [hardening_security, login_defs]

- name: "4.1 - Backup /etc/login.defs (only if changed)"
  ansible.builtin.copy:
    src: /etc/login.defs
    dest: "/etc/login.defs.{{ backup_date }}"
    remote_src: true
    owner: root
    group: root
    mode: "0644"
  when:
    - login_defs_stat.stat.exists
    - (ld_max.changed or ld_min.changed or ld_warn.changed or ld_len.changed)
  tags: [hardening_security, login_defs]

#####################################
# 4.2 Configurações SSH (guide-only, idempotente + backup só se mudar)
#####################################

- name: "4.2 - Stat /etc/ssh/sshd_config"
  ansible.builtin.stat:
    path: /etc/ssh/sshd_config
  register: sshd_conf_stat
  tags: [hardening_security, ssh]

- name: "4.2 - Configure SSH hardening options"
  ansible.builtin.lineinfile:
    path: /etc/ssh/sshd_config
    regexp: '^\s*#?\s*{{ item.key }}\b.*'
    line: "{{ item.key }} {{ item.value }}"
    create: false
  loop:
    - { key: "LoginGraceTime",      value: "0" }
    - { key: "ClientAliveInterval", value: "0" }
    - { key: "UseDNS",              value: "no" }
    - { key: "PermitRootLogin",     value: "no" }
  register: ssh_main
  when: sshd_conf_stat.stat.exists
  tags: [hardening_security, ssh]

- name: "4.2 - Ensure SSH Port 22"
  ansible.builtin.lineinfile:
    path: /etc/ssh/sshd_config
    regexp: '^\s*Port\s+22\b'
    line: "Port 22"
    insertafter: EOF
    create: false
  register: ssh_port_22
  when: sshd_conf_stat.stat.exists
  tags: [hardening_security, ssh]

- name: "4.2 - Ensure SSH Port 20022"
  ansible.builtin.lineinfile:
    path: /etc/ssh/sshd_config
    regexp: '^\s*Port\s+20022\b'
    line: "Port 20022"
    insertafter: EOF
    create: false
  register: ssh_port_20022
  when: sshd_conf_stat.stat.exists
  tags: [hardening_security, ssh]

- name: "4.2 - Backup /etc/ssh/sshd_config (only if changed)"
  ansible.builtin.copy:
    src: /etc/ssh/sshd_config
    dest: "/etc/ssh/sshd_config.{{ backup_date }}"
    remote_src: true
    owner: root
    group: root
    mode: "0600"
  when:
    - sshd_conf_stat.stat.exists
    - (ssh_main.changed or ssh_port_22.changed or ssh_port_20022.changed)
  tags: [hardening_security, ssh]

- name: "4.2 - Validate sshd config (only if changed)"
  ansible.builtin.command: sshd -t
  changed_when: false
  when:
    - sshd_conf_stat.stat.exists
    - (ssh_main.changed or ssh_port_22.changed or ssh_port_20022.changed)
  tags: [hardening_security, ssh]

- name: "4.2 - Restart sshd (only if changed)"
  ansible.builtin.systemd:
    name: sshd
    state: restarted
  when:
    - sshd_conf_stat.stat.exists
    - (ssh_main.changed or ssh_port_22.changed or ssh_port_20022.changed)
